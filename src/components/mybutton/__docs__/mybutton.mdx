## Usage

 ```ts
 import {MyButton} from "milibreria";

 function Example(){
  return (
      <MyButton
           text: "Button primary",
           txtColor: 'black',
           bgColor: 'orange',
           borderColor: 'green',
           border: '3',
           size: 'large',
           hoverBgColor: 'blue',
           hoverTxtColor: 'white',
           disabled: false,
           onClick: () => alert("Button Primary"),
      />
  );
 };

 export default Example;
 ```
